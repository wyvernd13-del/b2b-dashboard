// ─── GOOGLE SHEETS API ───────────────────────────────────────────────────────
async function sheetsGet(spreadsheetId, range) {
  const url = `https://sheets.googleapis.com/v4/spreadsheets/${spreadsheetId}/values/${encodeURIComponent(range)}?key=${API_KEY}`;
  const res = await fetch(url);
  if (!res.ok) throw new Error(`Sheets API error ${res.status}: ${await res.text()}`);
  const json = await res.json();
  return json.values || [];
}

async function getSheetNames(spreadsheetId) {
  const url = `https://sheets.googleapis.com/v4/spreadsheets/${spreadsheetId}?key=${API_KEY}&fields=sheets.properties.title`;
  const res = await fetch(url);
  if (!res.ok) throw new Error(`Sheets API error ${res.status}`);
  const json = await res.json();
  return (json.sheets || []).map(s => s.properties.title);
}

// ─── DATA PARSING ─────────────────────────────────────────────────────────────
function parseNum(v) {
  if (v===undefined || v===null || v==='') return 0;
  const n = parseFloat(String(v).replace(/[^0-9.-]/g,''));
  return isNaN(n) ? 0 : n;
}

function parseDate(v) {
  if (!v) return null;
  // Try ISO
  const d = new Date(v);
  if (!isNaN(d)) return d;
  // Try dd.mm.yyyy
  const parts = v.split('.');
  if (parts.length===3) return new Date(+parts[2], +parts[1]-1, +parts[0]);
  return null;
}

function monthKey(d) {
  if (!d) return null;
  const months = ['JAN','FEB','MAR','APR','MAY','JUN','JUL','AUG','SEP','OCT','NOV','DEC'];
  return months[d.getMonth()];
}

    const label = (row[0]||'').toString().trim().toLowerCase();

    // Monthly summary row
    const monthMatch = label.match(/monthly\s+(\w+)/i) || label.match(/^(january|february|march|april|may|june|july|august|september|october|november|december)$/i);
    if (monthMatch) {
      const mName = (monthMatch[1]||monthMatch[0]).toUpperCase().slice(0,3);
      monthly[mName] = {
        calls:      parseNum(row[2]),
        reached:    parseNum(row[3]),
        interested: parseNum(row[5]),
        sent:       parseNum(row[6]),
        talkTime:   row[8]||'',
      };
      continue;
    }

    // Weekly summary row
    if (label === 'weekly' || label.startsWith('week')) {
      weekly.push({
        calls:      parseNum(row[2]),
        reached:    parseNum(row[3]),
        interested: parseNum(row[5]),
        sent:       parseNum(row[6]),
      });
      continue;
    }

    // Daily row — label should be a date
    const dt = parseDate(row[0]);
    if (!dt || isNaN(dt)) continue;
    const calls = parseNum(row[2]);
    if (calls === 0 && parseNum(row[3]) === 0) continue; // skip empty days

    daily.push({
      date:       dt,
      dateStr:    dt.toISOString().slice(0,10),
      agent:      agentName,
      team,
      color,
      calls:      calls,
      reached:    parseNum(row[3]),
      interested: parseNum(row[5]),
      sent:       parseNum(row[6]),
      talkTime:   row[8]||'',
    });
  }

  return {daily, monthly, weekly};
}

// ─── MAIN FETCH ───────────────────────────────────────────────────────────────
async function fetchAllData() {
  setLoading(true, 'Fetching agent sheets…');

  const newDB = {daily:[], weekly:[], monthly:{}, agents:{}, lastFetch: new Date()};

  // Fetch each agent's sheet from the daily report spreadsheet
  const allWeeklyByAgent = {};

  for (let ai = 0; ai < AGENT_CONFIG.length; ai++) {
    const ag = AGENT_CONFIG[ai];
    setLoading(true, `Loading ${ag.name} (${ai+1}/${AGENT_CONFIG.length})…`);
    try {
      const rows = await sheetsGet(DAILY_SHEET_ID, `'${ag.sheet}'!A:J`);
      if (rows.length < 2) continue;
      const parsed = parseAgentSheet(rows, ag.name, ag.team, ag.color);

      newDB.daily.push(...parsed.daily);
      newDB.agents[ag.name] = {
        team: ag.team,
        color: ag.color,
        months: parsed.monthly,
        weekly: parsed.weekly,
      };
      allWeeklyByAgent[ag.name] = parsed.weekly;
    } catch(e) {
      console.warn(`Could not load ${ag.sheet}:`, e.message);
    }
  }

  // Sort daily by date desc
  newDB.daily.sort((a,b) => b.date - a.date);

  // Build monthly totals across all agents
  const allMonths = ['JAN','FEB','MAR','APR','MAY','JUN','JUL','AUG','SEP','OCT','NOV','DEC'];
  allMonths.forEach(m => {
    let calls=0,reached=0,interested=0,sent=0;
    Object.values(newDB.agents).forEach(ag => {
      const md = ag.months[m];
      if(md){ calls+=md.calls; reached+=md.reached; interested+=md.interested; sent+=md.sent; }
    });
    if(calls>0) newDB.monthly[m] = {calls,reached,interested,sent};
  });

  // Build weekly totals by aggregating daily data per week
  const weekMap = {};
  newDB.daily.forEach(d => {
    const date = d.date;
    // Get ISO week
    const tmp = new Date(date);
    tmp.setHours(0,0,0,0);
    tmp.setDate(tmp.getDate() + 3 - (tmp.getDay()+6)%7);
    const week1 = new Date(tmp.getFullYear(),0,4);
    const wn = 1 + Math.round(((tmp-week1)/86400000 - 3 + (week1.getDay()+6)%7)/7);
    const yr = tmp.getFullYear();
    const key = `${yr}-W${String(wn).padStart(2,'0')}`;
    if(!weekMap[key]) weekMap[key] = {wn, yr, calls:0, reached:0, interested:0, sent:0, dates:[]};
    weekMap[key].calls += d.calls;
    weekMap[key].reached += d.reached;
    weekMap[key].interested += d.interested;
    weekMap[key].sent += d.sent;
    weekMap[key].dates.push(date);
  });
  newDB.weekly = Object.entries(weekMap)
    .sort((a,b) => a[0].localeCompare(b[0]))
    .filter(([,w]) => w.calls > 5)
    .map(([key,w]) => {
      const dts = w.dates.sort((a,b)=>a-b);
      const fmt2 = d => `${d.getDate()} ${['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'][d.getMonth()]}`;
      return {
        label: `Wk ${w.wn}`,
        dateLabel: dts.length ? `${fmt2(dts[0])}` : '',
        calls: w.calls, reached: w.reached, interested: w.interested, sent: w.sent
      };
    });

  DB = newDB;

  // Populate agent filter dropdown
  const sel = document.getElementById('dailyAgentFilter');
  sel.innerHTML = '<option value="ALL">All agents</option>';
  Object.keys(DB.agents).sort().forEach(name => {
    sel.insertAdjacentHTML('beforeend', `<option value="${name}">${name}</option>`);
  });

  setSyncStatus(true);
  setLoading(false);
  renderPage();
}

// ─── PERIOD DATA ──────────────────────────────────────────────────────────────
function getPeriodAgentData(agName) {
  const ag = DB.agents[agName];
  if (!ag) return null;
  if (currentMonth === 'ALL') {
    let calls=0,reached=0,interested=0,sent=0;
    Object.values(ag.months).forEach(m=>{calls+=m.calls;reached+=m.reached;interested+=m.interested;sent+=m.sent;});
    return {team:ag.team,color:ag.color,calls,reached,interested,sent};
  }
  const m = ag.months[currentMonth];
  if (!m) return {team:ag.team,color:ag.color,calls:0,reached:0,interested:0,sent:0};
  return {team:ag.team,color:ag.color,...m};
}

function getPeriodTotal() {
  if (currentMonth === 'ALL') {
    let calls=0,reached=0,interested=0,sent=0;
    Object.values(DB.monthly).forEach(m=>{calls+=m.calls;reached+=m.reached;interested+=m.interested;sent+=m.sent;});
    return {calls,reached,interested,sent};
  }
  return DB.monthly[currentMonth] || {calls:0,reached:0,interested:0,sent:0};
}

    if(!a) return;
    Object.entries(a.months).forEach(([m,d])=>{
      csv += `${ag.name},${ag.team},${m},${d.calls},${d.reached},${pct(d.reached,d.calls)},${d.interested},${pct(d.interested,d.reached)},${d.sent}\n`;
    });
  });
  const blob = new Blob([csv],{type:'text/csv'});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href=url; a.download=`b2b_report_${new Date().toISOString().slice(0,10)}.csv`; a.click();
  URL.revokeObjectURL(url);
  toast('CSV exported!');
}

// ─── REFRESH ──────────────────────────────────────────────────────────────────
async function refreshAll() {
  const btn = document.getElementById('refreshBtn');
  btn.classList.add('spinning');
  try {
    await fetchAllData();
    toast('Data refreshed!');
  } catch(e) {
    setSyncStatus(false, 'Error');
    showError('Could not fetch data: ' + e.message + '. Check the API key and that the sheets are public.');
    toast('Refresh failed', true);
  } finally {
    btn.classList.remove('spinning');
  }
}

function toast(msg, err=false) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.style.background = err ? 'var(--danger)' : 'var(--accent2)';
  t.style.color = err ? '#fff' : '#0f1117';
  t.classList.add('show');
  setTimeout(()=>t.classList.remove('show'), 2500);
}
